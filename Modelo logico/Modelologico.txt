TABALA ALUMNOS
    ID_ALUMNO   ENTERO UNSIGNED NOT NULL PRIMARY KEY
    DNI         cadena de 9 aracteres NOT NULL UNIQUE
    NOMBRE      cadena de caracteres NOT NULL
    TELEFONO    ENTERO NOT NULL UNIQUE
    FECHA_NAC   DATE NOT NULL
    CURSO       ENTERO UNSIGNED NOT NULL
                FK REFERENCIA A LA TABLA CURSOS(ID_CURSO)
    AULA        ENTERO UNISGNED NOT NULL
                FK REFERENCIA A LA TABLA AULAS(ID_AULA)

TABLA CURSOS
    ID_CURSO    ENTERO NOT NULL PRIMARY KEY
    NOMBRE      cadena de caracteres NOT NULL UNIQUE
    DURACION    BYTE UNSIGNED NOT NULL
    NIVEL       ENUMERADO NOT NULL
    DELEGADO    ENTERO UNSIGNED NOT NULL
                FK REFERENCIA A LA TABLA ALUMNOS(ID_ALUMNOS)
    TUTOR       cadena de 9 caracteres NOT NULL
                FK REFERENCIA A LA TABLA PROFESORES(DNI)

TABLA AULAS GRADO: 5
    ID_AULA     ENTERO UNSIGNED NOT NULL PRIMARY KEY
    NOMBRE      NOT NULL
    PISO        BYTE NOT NULL
    AFORO       BYTE NOT NULL
    
TABLA PROFESORES
    DNI         cadena de 9 caracteres NOT NULL PRIMARY KEY
    NOMBRE      cadena de caracteres    NOT NULL
    SALARIO     FLOAT NOT NULL
    FECHA_NAC   DATE NULL

TABLA PROFESORES_CURSOS
    PROFESOR    cadena de 9 caracteres NOT NULL
                FK REFERENCIA A LA TABLA PROFESORES(DNI)
    CURSO       ENTERO UNSIGNED NOT NULL
                FK REFERENCIA A LA TABLA CURSOS(ID_CURSO)
    FECHA_MATRICULA DATE NOT NULL
    PRIMARY KEY (PROFESOR CURSO)
